# Find Curses (we need ncurses)
SET(CURSES_NEED_NCURSES TRUE)
FIND_PACKAGE(Curses REQUIRED)

# Find Format
FIND_PACKAGE(Format REQUIRED)

# Build our NUClear module
NUCLEAR_MODULE(INCLUDES ${CURSES_INCLUDE_DIRS} ${FORMAT_INCLUDE_DIRS}
               LIBRARIES ${CURSES_LIBRARIES} ${FORMAT_LIBRARIES})
